package com.example.italwebshop2;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Context;
import android.content.Intent;
import android.os.Bundle;
import android.widget.ArrayAdapter;
import android.widget.ListView;
import java.util.ArrayList;
import java.util.List;

public class Cart extends AppCompatActivity {

    private static final String EXTRA_CART_LIST = "com.example.italdiszkontwebshop.EXTRA_CART_LIST";
    private static List<String> cartList = new ArrayList<>();

    private ArrayAdapter<String> cartAdapter;

    public static Intent getIntent(Context context, List<String> cartItems) {
        Intent intent = new Intent(context, Cart.class);
        intent.putStringArrayListExtra(EXTRA_CART_LIST, new ArrayList<>(cartItems));
        return intent;
    }

    public static List<String> getCartList() {
        return cartList;
    }

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_cart);

        ListView cartListView = findViewById(R.id.cartListView);

        cartList = getIntent().getStringArrayListExtra(EXTRA_CART_LIST);
        cartAdapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, cartList);
        cartListView.setAdapter(cartAdapter);
    }
}
