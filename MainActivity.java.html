package com.example.italwebshop2;

import static com.example.italwebshop2.R.id.drinkListView;

import androidx.appcompat.app.AppCompatActivity;
import android.content.Intent;
import android.os.Bundle;
import android.view.View;
import android.widget.AdapterView;
import android.widget.ArrayAdapter;
import android.widget.Button;
import android.widget.ListView;
import android.widget.Toast;
import java.util.ArrayList;
import java.util.HashMap;
import java.util.List;
import java.util.Map;

public class MainActivity extends AppCompatActivity {

    private Map<String, List<String>> drinkMap;
    private List<String> categoryList;
    private List<String> cartList;
    private ArrayAdapter<String> categoryAdapter;
    private ArrayAdapter<String> drinkAdapter;
    private ArrayAdapter<String> cartAdapter;

    @Override
    protected void onCreate(Bundle savedInstanceState) {
        super.onCreate(savedInstanceState);
        setContentView(R.layout.activity_main);

        ListView categoryListView = findViewById(R.id.categoryListView);
        ListView drinkListView = findViewById(R.id.drinkListView);
        Button cartButton = findViewById(R.id.cartButton);

        categoryList = new ArrayList<>();
        drinkMap = new HashMap<>();
        cartList = new ArrayList<>();
        categoryAdapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, categoryList);
        drinkAdapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, new ArrayList<>());
        cartAdapter = new ArrayAdapter<>(this, android.R.layout.simple_list_item_1, cartList);

        categoryListView.setAdapter(categoryAdapter);
        drinkListView.setAdapter(drinkAdapter);
        categoryList.add("Alkoholos");
        categoryList.add("Alkoholmentes");
        categoryAdapter.notifyDataSetChanged();
        List<String> alcoholicDrinks = new ArrayList<>();
        alcoholicDrinks.add("Sör (Borsodi)");
        alcoholicDrinks.add("Bor (Nyakas Pince)");
        alcoholicDrinks.add("Pálinka");
        alcoholicDrinks.add("Likőr");
        alcoholicDrinks.add("Vodka");
        alcoholicDrinks.add("Tequila");
        alcoholicDrinks.add("Gin");
        drinkMap.put("Alkoholos", alcoholicDrinks);

        List<String> nonAlcoholicDrinks = new ArrayList<>();
        nonAlcoholicDrinks.add("Coca Cola");
        nonAlcoholicDrinks.add("Ice Tea");
        nonAlcoholicDrinks.add("Almalé");
        nonAlcoholicDrinks.add("Hell");
        drinkMap.put("Alkoholmentes", nonAlcoholicDrinks);

        categoryListView.setOnItemClickListener(new AdapterView.OnItemClickListener() {
            @Override
            public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
                String category = categoryList.get(position);
                List<String> drinks = drinkMap.get(category);
                drinkAdapter.clear();
                drinkAdapter.addAll(drinks);
                drinkAdapter.notifyDataSetChanged();
            }
        });

        drinkListView.setOnItemClickListener(new AdapterView.OnItemClickListener() {
            @Override
            public void onItemClick(AdapterView<?> parent, View view, int position, long id) {
                String drink = drinkAdapter.getItem(position);
                cartList.add(drink);
                cartAdapter.notifyDataSetChanged();
                Toast.makeText(MainActivity.this, drink + " added to cart.", Toast.LENGTH_SHORT).show();
            }
        });

        cartButton.setOnClickListener(new View.OnClickListener() {
            @Override
            public void onClick(View v) {
                // Start CartActivity
                startActivity(Cart.getIntent(MainActivity.this, cartList));
            }
        });
    }
}
